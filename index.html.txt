/* ── API Helper ── */
async function api(method,path,body){
  var opts={method:method,headers:{'Content-Type':'application/json'},credentials:'same-origin'};
  if(body)opts.body=JSON.stringify(body);
  var controller=new AbortController();
  var timeout=setTimeout(function(){controller.abort();},30000);
  try{
    var res=await fetch(path,Object.assign({},opts,{signal:controller.signal}));
    clearTimeout(timeout);
    var data=await res.json();
    if(!res.ok){
      if(res.status===401){showLogin();throw new Error('Session expired');}
      throw new Error(data.error||'Request failed');
    }
    return data;
  }catch(e){
    clearTimeout(timeout);
    if(e.name==='AbortError')throw new Error('Server waking up, try again');
    throw e;
  }
}

/* ── Save Transaction ── */
async function saveTransaction(){
  var amount=parseFloat(document.getElementById('amount-input').value);
  var date=document.getElementById('date-input').value;
  var note=document.getElementById('note-input').value.trim();
  var errEl=document.getElementById('amount-error');
  if(!amount||amount<=0){errEl.classList.remove('hidden');return;}
  errEl.classList.add('hidden');
  if(!date)date=todayStr();
  if(!state.selectedCategory){showToast('Please select a category','error');return;}
  var txn={date:date,type:state.currentType,category:state.selectedCategory,amount:amount,note:note};
  var editRow=document.getElementById('edit-row').value;
  var btn=document.getElementById('save-btn');
  var btnText=document.getElementById('save-btn-text');
  var btnLoader=document.getElementById('save-btn-loader');
  btn.disabled=true;btnText.classList.add('hidden');btnLoader.classList.remove('hidden');
  try{
    if(editRow){await api('PUT','/api/transactions/'+editRow,txn);}
    else{await api('POST','/api/transactions',txn);}
    btn.disabled=false;btnText.classList.remove('hidden');btnLoader.classList.add('hidden');
    showToast(editRow?'Transaction updated':'Transaction added','success');
    resetForm();
    await refreshData();
    navigateTo('home');
  }catch(e){
    btn.disabled=false;btnText.classList.remove('hidden');btnLoader.classList.add('hidden');
    showToast('Error: '+e.message,'error');
  }

    showLogin();
  }
